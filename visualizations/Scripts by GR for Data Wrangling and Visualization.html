TOD <- cut(all_data$HofD.steps,breaks = c(0,4,10,17,24),labels = c('Too early?','Morning','Mid-day','Evening'),right=T)
TOD[is.na(TOD)] <- factor('Too early?',levels = levels(TOD))

all_data$id <- factor(all_data$id)
all_data <- filter(all_data, tot_steps > 0)
